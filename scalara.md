```sql
CREATE TABLE IF NOT EXIST user(
id INTEGER,
name VARCHAR,
trophies INTEGER,
arena INTEGER,
tower_lvl INTEGER,
clan_id INTEGER,
coloda_number INTEGER,
now_coloda INTEGER,
amount_of_cards INTEGER,
gold INTEGER
);

CREATE TABLE IF NOT EXIST cards(
varvar VARCHAR,
bochka VARCHAR,
brevno VARCHAR
)
CREATE TABLE IF NOT EXIST stats brevno(
mana INTEGER,
dmg INTEGER,
spawntime INTEGER,
dmg_on_tower INTEGER
)
```
